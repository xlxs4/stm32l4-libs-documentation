1096 {
1097  uint32_t groups = 0UL;
1098  uint32_t idx;
1099 
1100  for (idx = 0UL; idx < (uint32_t)TSC_NB_OF_GROUPS; idx++)
1101  {

1103  {
1104  groups |= (1UL << idx);
