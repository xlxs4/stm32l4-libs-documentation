147 ErrorStatus LL_I2C_Init(I2C_TypeDef *I2Cx, LL_I2C_InitTypeDef *I2C_InitStruct)
148 {
149  /* Check the I2C Instance I2Cx */
150  assert_param(IS_I2C_ALL_INSTANCE(I2Cx));
151 
152  /* Check the I2C parameters from I2C_InitStruct */
153  assert_param(IS_LL_I2C_PERIPHERAL_MODE(I2C_InitStruct->PeripheralMode));
154  assert_param(IS_LL_I2C_ANALOG_FILTER(I2C_InitStruct->AnalogFilter));
155  assert_param(IS_LL_I2C_DIGITAL_FILTER(I2C_InitStruct->DigitalFilter));
156  assert_param(IS_LL_I2C_OWN_ADDRESS1(I2C_InitStruct->OwnAddress1));
157  assert_param(IS_LL_I2C_TYPE_ACKNOWLEDGE(I2C_InitStruct->TypeAcknowledge));
158  assert_param(IS_LL_I2C_OWN_ADDRSIZE(I2C_InitStruct->OwnAddrSize));
159 
160  /* Disable the selected I2Cx Peripheral */
161  LL_I2C_Disable(I2Cx);
162 
163  /*---------------------------- I2Cx CR1 Configuration ------------------------
164  * Configure the analog and digital noise filters with parameters :
165  * - AnalogFilter: I2C_CR1_ANFOFF bit
166  * - DigitalFilter: I2C_CR1_DNF[3:0] bits
167  */
168  LL_I2C_ConfigFilters(I2Cx, I2C_InitStruct->AnalogFilter, I2C_InitStruct->DigitalFilter);
169 
170  /*---------------------------- I2Cx TIMINGR Configuration --------------------
171  * Configure the SDA setup, hold time and the SCL high, low period with parameter :
172  * - Timing: I2C_TIMINGR_PRESC[3:0], I2C_TIMINGR_SCLDEL[3:0], I2C_TIMINGR_SDADEL[3:0],
173  * I2C_TIMINGR_SCLH[7:0] and I2C_TIMINGR_SCLL[7:0] bits
174  */
175  LL_I2C_SetTiming(I2Cx, I2C_InitStruct->Timing);
176 
177  /* Enable the selected I2Cx Peripheral */
178  LL_I2C_Enable(I2Cx);
179 
180  /*---------------------------- I2Cx OAR1 Configuration -----------------------
181  * Disable, Configure and Enable I2Cx device own address 1 with parameters :
182  * - OwnAddress1: I2C_OAR1_OA1[9:0] bits
183  * - OwnAddrSize: I2C_OAR1_OA1MODE bit
184  */
185  LL_I2C_DisableOwnAddress1(I2Cx);
186  LL_I2C_SetOwnAddress1(I2Cx, I2C_InitStruct->OwnAddress1, I2C_InitStruct->OwnAddrSize);
187 
188  /* OwnAdress1 == 0 is reserved for General Call address */
189  if (I2C_InitStruct->OwnAddress1 != 0U)
190  {
191  LL_I2C_EnableOwnAddress1(I2Cx);
192  }
193 
194  /*---------------------------- I2Cx MODE Configuration -----------------------
195  * Configure I2Cx peripheral mode with parameter :
196  * - PeripheralMode: I2C_CR1_SMBDEN and I2C_CR1_SMBHEN bits
197  */
198  LL_I2C_SetMode(I2Cx, I2C_InitStruct->PeripheralMode);
199 
200  /*---------------------------- I2Cx CR2 Configuration ------------------------
201  * Configure the ACKnowledge or Non ACKnowledge condition
202  * after the address receive match code or next received byte with parameter :
203  * - TypeAcknowledge: I2C_CR2_NACK bit
204  */
205  LL_I2C_AcknowledgeNextData(I2Cx, I2C_InitStruct->TypeAcknowledge);
206 
207  return SUCCESS;
208 }
